class Node():
    def __init__(self, data):
        self.data = data
        self.next = None
class LinkedList():
    def __init__(self):
        self.head = Node(None)
    def insertAtLastNode(self, newNode):
        curNode = self.head
        while curNode.next != None:
            curNode = curNode.next
        curNode.next = newNode
class Queue():
    def __init__(self, queueCapacity):
        assert queueCapacity > 0, "QueueCapacity需大於0"
        self.capacity = queueCapacity
        self.myQueue = [None] * self.capacity
        self.front = -1
        self.rear = -1
    def isEmpty(self):
        if self.front == self.rear:
            return True
        else:
            return False
    def Push(self, value):
        self.rear += 1
        self.myQueue[self.rear] = value
    def Pop(self):
        self.front += 1
        return self.myQueue[self.front]
